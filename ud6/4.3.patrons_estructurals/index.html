interface PizzaAdapter {

class PizzaXMLAdapter implements PizzaAdapter {
    @Override
    public JSONObject adaptarAJSON(String xml) {
        try {
            // Convertir XML a JSONObject
            DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
            DocumentBuilder builder = factory.newDocumentBuilder();
            InputStream inputStream = new ByteArrayInputStream(xml.getBytes());
            Document doc = builder.parse(inputStream);
            doc.getDocumentElement().normalize();

            // Suposant que l'XML t茅 una estructura similar a:
            // <pizza>

            //    <preu>8.5</preu>

            JSONObject pizzaJSON = new JSONObject();
            for (int i = 0; i < nodeList.getLength(); i++) {

                    Element element = (Element) node;

                    // Creem el JSON a partir de l'XML
                    pizzaJSON.put("nom", nom);
                    pizzaJSON.put("preu", preu);
                }
            }
            return pizzaJSON;
        } catch (Exception e) {

        }
        return null;
    }
}
